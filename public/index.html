Security Monitoring

VictoriaLogs — Security Queries
# Failed SSH attempts
_stream:{service="sshd"} AND
_msg:"Failed password"

# Sudo abuse
_stream:{service="sudo"} AND
_msg:"NOT in sudoers"

# Firewall drops
_stream:{service="nftables"} AND
_msg:"DROP"
VictoriaMetrics — Security Metrics
# Rate of failed auth attempts
rate(auth_failures_total[5m])

# SSH connection rate spikes
rate(ssh_connections_total[1m]) > 10

# Firewall block rate
rate(nftables_drops_total[5m])
